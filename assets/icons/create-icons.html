图标生成工具

打开浏览器控制台，运行以下代码生成图标：

// 创建canvas并绘制图标
function createIcon(size) {
  const canvas = document.createElement('canvas');
  canvas.width = size;
  canvas.height = size;
  const ctx = canvas.getContext('2d');
  
  // 背景
  ctx.fillStyle = '#2196f3';
  ctx.fillRect(0, 0, size, size);
  
  // 绘制对话气泡
  ctx.fillStyle = 'white';
  ctx.beginPath();
  ctx.arc(size * 0.3, size * 0.35, size * 0.15, 0, Math.PI * 2);
  ctx.fill();
  
  ctx.beginPath();
  ctx.arc(size * 0.7, size * 0.65, size * 0.15, 0, Math.PI * 2);
  ctx.fill();
  
  // 绘制AI符号
  ctx.fillStyle = '#2196f3';
  ctx.font = `bold ${size * 0.3}px Arial`;
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText('AI', size * 0.5, size * 0.5);
  
  return canvas.toDataURL('image/png');
}

// 下载图标
function downloadIcon(size, filename) {
  const dataUrl = createIcon(size);
  const link = document.createElement('a');
  link.download = filename;
  link.href = dataUrl;
  link.click();
}

// 生成所有尺寸
downloadIcon(16, 'icon16.png');
downloadIcon(48, 'icon48.png');
downloadIcon(128, 'icon128.png');
